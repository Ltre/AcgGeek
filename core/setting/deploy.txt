
# clone方式更新

mkdir /home/wwwbackup/
cd /home/wwwroot/
tar -cf /home/wwwbackup/acggeek.com.tar  acggeek.com/
git clone https://github.com/Ltre/AcgGeek.git
rm -f -r AcgGeek/core/data/
mv acggeek.com/core/data/  AcgGeek/core/data/
mv acggeek.com/  acggeek.com.trash/
mv AcgGeek/ acggeek.com/
chmod -R 767 acggeek.com/core/data/
rm -f -r acggeek.com.trash/
